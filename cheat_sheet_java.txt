+-------------------------------------------------------+

	System.getProperty("java.library.path") where native classes are loaded from. meaning .dll or .so

	System.getProperty("java.class.path") where pure java classes are loaded from 
	Because JsonUtil.class is Java bytecode, not a native DLL.

+-------------------------------------------------------+
|									======1_of_5======	|
|	FIND_MENTIONS_OF_CLASS_IN_JAR						|
|														|
+-------------------------------------------------------+
	##linux
	for i in *.jar; do jar -tvf "$i" | grep -Hsi ClassName && echo "$i"; done
	
	##windows
	for /R %G in (*.jar) do @jar -tvf "%G" | find "ClassName" > NUL && echo %G

|									======1_of_5======	|
|	FIND_IMPLEMENTATION_OF_CLASS_IN_JAR					|
|														|
+-------------------------------------------------------+	


	##linux
	find /path/to/WEB-INF/lib-connectors -name "*.jar" -type f | while read -r f; do
	  if jar tf "$f" | grep -qi "connector/common/JsonUtil.class"; then
		echo "FOUND IN $f"
	  fi
	done

	##windows
	for %f in (*.jar) do @jar tf "%f" | findstr /i "connector/common/JsonUtil.class" && echo FOUND IN %f
