+-------------------------------------------------------+
|														|
|	CREATE AND SWITCH TO A NEW BRANCH IN GIT			|
|														|
+-------------------------------------------------------+
    git checkout -b new-branch-name


+-------------------------------------------------------+
|														|
|	VERIFY IF YOU ARE WORKING ON THAT BRANCH			|
|														|
+-------------------------------------------------------+
	git branch


+-------------------------------------------------------+
|														|
|	PUSH CREATED TO A REMOTE REPOSITORY					|
|														|
+-------------------------------------------------------+
	git push origin new-branch-name
